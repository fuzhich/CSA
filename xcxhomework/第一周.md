八.

```go
package main

import "fmt"

func main() {
    var (
       x int
       y int
    )
    var sum int
    var (
       big   int
       small int
    )
    for {
       m := make([]int, 0)
       fmt.Scanf("%d %d", &x, &y)
       if x <= 0 || y <= 0 {
          break
       }
       if x < y {
          small = x
          big = y
       } else {
          small = y
          big = x
       }
       for i := small; i <= big; i++ {
          m = append(m, i)
       }
       sum = 0
       for _, value := range m {
          fmt.Printf("%d ", value)
          sum += value
       }
       fmt.Printf("Sum=%d\n", sum)
    }
}
```

九.

```go
package main

import "fmt"

func main() {
    var n int
    fmt.Scanf("%d", &n)
    if 1 <= n && n <= 10 {
       fmt.Println(factorial(n))
    }
}
func factorial(n int) int {
    if n == 0 || n == 1 {
       return 1
    }
    return factorial(n-1) * n
}
```

十.

```go
package main

import "fmt"

func main() {
    var (
       a int
       b int
    )
    fmt.Scanf("%d %d", &a, &b)
    fmt.Println(lcm(a, b))
}

func lcm(a, b int) int {
    return (a * b) / gcd(a, b)
}

func gcd(a, b int) int {
    for b != 0 {
       a, b = b, a%b
    }
    return a
}
```

猜数

```go
package main

import (
    "fmt"
    "math/rand"
    "time"
)

func main() {
    maxNum := 100
    var guess int
    for {
       rand.Seed(time.Now().UnixNano())
       secretNumber := rand.Intn(maxNum)
       fmt.Println("Please input your guess")
       for {
          _, err := fmt.Scanf("%d", &guess)
          if err != nil {
             fmt.Println("Invalid input. Please enter an integer value")
             return
          }
          if guess < secretNumber {
             fmt.Println("Your guess is small than the secretNumber")
          } else if guess > secretNumber {
             fmt.Println("Your guess is bigger than the secretNumber")
          } else {
             fmt.Println("Your guess is equal to secretNumber")
             fmt.Println("The secretNumber is ", secretNumber)
             break
          }
       }
    }
}
```
